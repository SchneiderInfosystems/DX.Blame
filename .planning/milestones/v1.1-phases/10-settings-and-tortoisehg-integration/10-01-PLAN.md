   - Insert a case block before `if not ScanForVCS(...)`:
     ```
     case BlameSettings.VCSPreference of
       vpGit:
       begin
         Result := TGitProvider.Create;
         if Result.FindExecutable = '' then
         begin
           Result := nil;
           Exit;
         end;
         ARepoRoot := Result.FindRepoRoot(AProjectPath);
         if ARepoRoot = '' then
           Result := nil;
         Exit;
       end;
       vpMercurial:
       begin
         Result := THgProvider.Create;
         if Result.FindExecutable = '' then
         begin
           Result := nil;
           Exit;
         end;
         ARepoRoot := Result.FindRepoRoot(AProjectPath);
         if ARepoRoot = '' then
           Result := nil;
         Exit;
       end;
     end;
     // vpAuto falls through to existing auto-detection
     ```
   - Add `DX.Blame.Settings` to the implementation uses clause (already present).

IMPORTANT: The forced provider path still validates executable existence and repo root, returning nil on failure (same as auto-detect). This means forcing Git on an Hg-only repo returns nil, which is correct.
